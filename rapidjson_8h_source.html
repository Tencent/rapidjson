140 //! Endianness of the machine.
141 /*!
142  \def RAPIDJSON_ENDIAN
143  \ingroup RAPIDJSON_CONFIG
144 
145  GCC 4.6 provided macro for detecting endianness of the target machine. But other
146  compilers may not have this. User can define RAPIDJSON_ENDIAN to either
147  \ref RAPIDJSON_LITTLEENDIAN or \ref RAPIDJSON_BIGENDIAN.
148 
149  Default detection implemented with reference to
150  \li https://gcc.gnu.org/onlinedocs/gcc-4.6.0/cpp/Common-Predefined-Macros.html
151  \li http://www.boost.org/doc/libs/1_42_0/boost/detail/endian.hpp
152 */
153 #ifndef RAPIDJSON_ENDIAN
154 // Detect with GCC 4.6's macro
155 # ifdef __BYTE_ORDER__
156 # if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
157 # define RAPIDJSON_ENDIAN RAPIDJSON_LITTLEENDIAN
158 # elif __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__
159 # define RAPIDJSON_ENDIAN RAPIDJSON_BIGENDIAN
160 # else
161 # error Unknown machine endianess detected. User needs to define RAPIDJSON_ENDIAN.
162 # endif // __BYTE_ORDER__
163 // Detect with GLIBC's endian.h
164 # elif defined(__GLIBC__)
165 # include <endian.h>
166 # if (__BYTE_ORDER == __LITTLE_ENDIAN)
167 # define RAPIDJSON_ENDIAN RAPIDJSON_LITTLEENDIAN
168 # elif (__BYTE_ORDER == __BIG_ENDIAN)
169 # define RAPIDJSON_ENDIAN RAPIDJSON_BIGENDIAN
170 # else
171 # error Unknown machine endianess detected. User needs to define RAPIDJSON_ENDIAN.
172 # endif // __GLIBC__
173 // Detect with _LITTLE_ENDIAN and _BIG_ENDIAN macro
174 # elif defined(_LITTLE_ENDIAN) && !defined(_BIG_ENDIAN)
175 # define RAPIDJSON_ENDIAN RAPIDJSON_LITTLEENDIAN
176 # elif defined(_BIG_ENDIAN) && !defined(_LITTLE_ENDIAN)
177 # define RAPIDJSON_ENDIAN RAPIDJSON_BIGENDIAN
178 // Detect with architecture macros
179 # elif defined(__sparc) || defined(__sparc__) || defined(_POWER) || defined(__powerpc__) || defined(__ppc__) || defined(__hpux) || defined(__hppa) || defined(_MIPSEB) || defined(_POWER) || defined(__s390__)
180 # define RAPIDJSON_ENDIAN RAPIDJSON_BIGENDIAN
181 # elif defined(__i386__) || defined(__alpha__) || defined(__ia64) || defined(__ia64__) || defined(_M_IX86) || defined(_M_IA64) || defined(_M_ALPHA) || defined(__amd64) || defined(__amd64__) || defined(_M_AMD64) || defined(__x86_64) || defined(__x86_64__) || defined(_M_X64) || defined(__bfin__)
182 # define RAPIDJSON_ENDIAN RAPIDJSON_LITTLEENDIAN
183 # elif defined(RAPIDJSON_DOXYGEN_RUNNING)
184 # define RAPIDJSON_ENDIAN
185 # else
186 # error Unknown machine endianess detected. User needs to define RAPIDJSON_ENDIAN.
187 # endif
188 #endif // RAPIDJSON_ENDIAN

472  \brief Concept for reading and writing characters.
473 
474  For read-only stream, no need to implement PutBegin(), Put(), Flush() and PutEnd().
475 
476  For write-only stream, only need to implement Put() and Flush().

479 concept Stream {
480  typename Ch; //!< Character type of the stream.
481 
482  //! Read the current character from stream without moving the read cursor.
483  Ch Peek() const;
484 
485  //! Read the current character from stream and moving the read cursor to next character.
486  Ch Take();
487 
488  //! Get the current read cursor.
489  //! \return Number of characters read from start.
490  size_t Tell();
491 
492  //! Begin writing operation at the current read pointer.
493  //! \return The begin writer pointer.
494  Ch* PutBegin();
495 
496  //! Write a character.
497  void Put(Ch c);
498 
499  //! Flush the buffer.
500  void Flush();
501 
502  //! End the writing operation.

536 //! Read-only string stream.
537 /*! \note implements Stream concept
538 */
539 template <typename Encoding>
540 struct GenericStringStream {
541  typedef typename Encoding::Ch Ch;
542 
543  GenericStringStream(const Ch *src) : src_(src), head_(src) {}
544 
545  Ch Peek() const { return *src_; }
546  Ch Take() { return *src_++; }
547  size_t Tell() const { return static_cast<size_t>(src_ - head_); }
548 
549  Ch* PutBegin() { RAPIDJSON_ASSERT(false); return 0; }
550  void Put(Ch) { RAPIDJSON_ASSERT(false); }
551  void Flush() { RAPIDJSON_ASSERT(false); }
552  size_t PutEnd(Ch*) { RAPIDJSON_ASSERT(false); return 0; }
553 
554  const Ch* src_; //!< Current read position.
555  const Ch* head_; //!< Original head of the string.
556 };

567 // InsituStringStream
568 
569 //! A read-write string stream.
570 /*! This string stream is particularly designed for in-situ parsing.
571  \note implements Stream concept
572 */
573 template <typename Encoding>
574 struct GenericInsituStringStream {
575  typedef typename Encoding::Ch Ch;
576 
577  GenericInsituStringStream(Ch *src) : src_(src), dst_(0), head_(src) {}
578 
579  // Read
580  Ch Peek() { return *src_; }
581  Ch Take() { return *src_++; }
582  size_t Tell() { return static_cast<size_t>(src_ - head_); }
583 
584  // Write
585  void Put(Ch c) { RAPIDJSON_ASSERT(dst_ != 0); *dst_++ = c; }
586 
587  Ch* PutBegin() { return dst_ = src_; }
588  size_t PutEnd(Ch* begin) { return static_cast<size_t>(dst_ - begin); }
589  void Flush() {}
590 
591  Ch* Push(size_t count) { Ch* begin = dst_; dst_ += count; return begin; }
592  void Pop(size_t count) { dst_ -= count; }
593 
594  Ch* src_;
595  Ch* dst_;
596  Ch* head_;
597 };
598

600 struct StreamTraits<GenericInsituStringStream<Encoding> > {
601  enum { copyOptimization = 1 };
