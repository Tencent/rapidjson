34 //! Represents an in-memory output stream.
35 /*!
36  \tparam Encoding Encoding of the stream.
37  \tparam Allocator type for allocating memory buffer.
38  \note implements Stream concept
39 */
40 template <typename Encoding, typename Allocator = CrtAllocator>
41 class GenericStringBuffer {
42 public:
43  typedef typename Encoding::Ch Ch;
44 
45  GenericStringBuffer(Allocator* allocator = 0, size_t capacity = kDefaultCapacity) : stack_(allocator, capacity) {}
46 
47 #if RAPIDJSON_HAS_CXX11_RVALUE_REFS
48  GenericStringBuffer(GenericStringBuffer&& rhs) : stack_(std::move(rhs.stack_)) {}
49  GenericStringBuffer& operator=(GenericStringBuffer&& rhs) {
50  if (&rhs != this)
51  stack_ = std::move(rhs.stack_);
52  return *this;
53  }
54 #endif
55 
56  void Put(Ch c) { *stack_.template Push<Ch>() = c; }
57  void Flush() {}
58 
59  void Clear() { stack_.Clear(); }
60  void ShrinkToFit() {
61  // Push and pop a null terminator. This is safe.
62  *stack_.template Push<Ch>() = '\0';
63  stack_.ShrinkToFit();
64  stack_.template Pop<Ch>(1);
65  }
66  Ch* Push(size_t count) { return stack_.template Push<Ch>(count); }
67  void Pop(size_t count) { stack_.template Pop<Ch>(count); }
68 
69  const Ch* GetString() const {
70  // Push and pop a null terminator. This is safe.
71  *stack_.template Push<Ch>() = '\0';
72  stack_.template Pop<Ch>(1);
73 
74  return stack_.template Bottom<Ch>();
75  }
76 
77  size_t GetSize() const { return stack_.GetSize(); }
78 
79  static const size_t kDefaultCapacity = 256;
80  mutable internal::Stack<Allocator> stack_;
81 
82 private:
83  // Prohibit copy constructor & assignment operator.
84  GenericStringBuffer(const GenericStringBuffer&);
85  GenericStringBuffer& operator=(const GenericStringBuffer&);
86 };

91 //! Implement specialized version of PutN() with memset() for better performance.
92 template<>
93 inline void PutN(GenericStringBuffer<UTF8<> >& stream, char c, size_t n) {
94  std::memset(stream.stack_.Push<char>(n), c, n * sizeof(c));
