31 /*! \class rapidjson::Allocator
32  \brief Concept for allocating, resizing and freeing memory block.
33 
34  Note that Malloc() and Realloc() are non-static but Free() is static.
35 
36  So if an allocator need to support Free(), it needs to put its pointer in
37  the header of memory block.
38 
39 \code
40 concept Allocator {
41  static const bool kNeedFree; //!< Whether this allocator needs to call Free().
42 
43  // Allocate a memory block.
44  // \param size of the memory block in bytes.
45  // \returns pointer to the memory block.
46  void* Malloc(size_t size);
47 
48  // Resize a memory block.
49  // \param originalPtr The pointer to current memory block. Null pointer is permitted.
50  // \param originalSize The current size in bytes. (Design issue: since some allocator may not book-keep this, explicitly pass to it can save memory.)
51  // \param newSize the new size in bytes.
52  void* Realloc(void* originalPtr, size_t originalSize, size_t newSize);
53 
54  // Free a memory block.
55  // \param pointer to the memory block. Null pointer is permitted.
56  static void Free(void *ptr);
57 };
58 \endcode
59 */
60

63  \ingroup RAPIDJSON_CONFIG
64  \brief User-defined kDefaultChunkCapacity definition.
65 
66  User can define this as any \c size that is a power of 2.
67 */

77 //! C-runtime library allocator.
78 /*! This class is just wrapper for standard C library memory routines.
79  \note implements Allocator concept
80 */
81 class CrtAllocator {
82 public:
83  static const bool kNeedFree = true;
84  void* Malloc(size_t size) {
85  if (size) // behavior of malloc(0) is implementation defined.
86  return RAPIDJSON_MALLOC(size);
87  else
88  return NULL; // standardize to returning NULL.
89  }
90  void* Realloc(void* originalPtr, size_t originalSize, size_t newSize) {
91  (void)originalSize;
92  if (newSize == 0) {
93  RAPIDJSON_FREE(originalPtr);
94  return NULL;
95  }
96  return RAPIDJSON_REALLOC(originalPtr, newSize);
97  }
98  static void Free(void *ptr) RAPIDJSON_NOEXCEPT { RAPIDJSON_FREE(ptr); }
99 
100  bool operator==(const CrtAllocator&) const RAPIDJSON_NOEXCEPT {
101  return true;
102  }
103  bool operator!=(const CrtAllocator&) const RAPIDJSON_NOEXCEPT {
104  return false;
105  }
106 };

108 ///////////////////////////////////////////////////////////////////////////////
109 // MemoryPoolAllocator
110 
111 //! Default memory allocator used by the parser and DOM.
112 /*! This allocator allocate memory blocks from pre-allocated memory chunks.
113 
114  It does not free memory blocks. And Realloc() only allocate new memory.
115 
116  The memory chunks are allocated by BaseAllocator, which is CrtAllocator by default.
117 
118  User may also supply a buffer as the first chunk.
119 
120  If the user-buffer is full then additional chunks are allocated by BaseAllocator.
121 
122  The user-buffer is not deallocated by this allocator.
123 
124  \tparam BaseAllocator the allocator type for allocating memory chunks. Default is CrtAllocator.
125  \note implements Allocator concept
126 */
127 template <typename BaseAllocator = CrtAllocator>
128 class MemoryPoolAllocator {
129  //! Chunk header for perpending to each chunk.
130  /*! Chunks are stored as a singly linked list.
131  */
132  struct ChunkHeader {
133  size_t capacity; //!< Capacity of the chunk in bytes (excluding the header itself).
134  size_t size; //!< Current size of allocated memory in bytes.
135  ChunkHeader *next; //!< Next chunk in the linked list.
136  };
137 
138  struct SharedData {
139  ChunkHeader *chunkHead; //!< Head of the chunk linked-list. Only the head chunk serves allocation.
140  BaseAllocator* ownBaseAllocator; //!< base allocator created by this object.
141  size_t refcount;
142  bool ownBuffer;
143  };
144 
145  static const size_t SIZEOF_SHARED_DATA = RAPIDJSON_ALIGN(sizeof(SharedData));
146  static const size_t SIZEOF_CHUNK_HEADER = RAPIDJSON_ALIGN(sizeof(ChunkHeader));
147 
148  static inline ChunkHeader *GetChunkHead(SharedData *shared)
149  {
150  return reinterpret_cast<ChunkHeader*>(reinterpret_cast<uint8_t*>(shared) + SIZEOF_SHARED_DATA);
151  }
152  static inline uint8_t *GetChunkBuffer(SharedData *shared)
153  {
154  return reinterpret_cast<uint8_t*>(shared->chunkHead) + SIZEOF_CHUNK_HEADER;
155  }
156 
157  static const size_t kDefaultChunkCapacity = RAPIDJSON_ALLOCATOR_DEFAULT_CHUNK_CAPACITY; //!< Default chunk capacity.
158 
159 public:
160  static const bool kNeedFree = false; //!< Tell users that no need to call Free() with this allocator. (concept Allocator)
161 
162  //! Constructor with chunkSize.
163  /*! \param chunkSize The size of memory chunk. The default is kDefaultChunkSize.
164  \param baseAllocator The allocator for allocating memory chunks.
165  */
166  explicit
167  MemoryPoolAllocator(size_t chunkSize = kDefaultChunkCapacity, BaseAllocator* baseAllocator = 0) :
168  chunk_capacity_(chunkSize),
169  baseAllocator_(baseAllocator ? baseAllocator : RAPIDJSON_NEW(BaseAllocator)()),
170  shared_(static_cast<SharedData*>(baseAllocator_ ? baseAllocator_->Malloc(SIZEOF_SHARED_DATA + SIZEOF_CHUNK_HEADER) : 0))
171  {
172  RAPIDJSON_ASSERT(baseAllocator_ != 0);
173  RAPIDJSON_ASSERT(shared_ != 0);
174  if (baseAllocator) {
175  shared_->ownBaseAllocator = 0;
176  }
177  else {
178  shared_->ownBaseAllocator = baseAllocator_;
179  }
180  shared_->chunkHead = GetChunkHead(shared_);
181  shared_->chunkHead->capacity = 0;
182  shared_->chunkHead->size = 0;
183  shared_->chunkHead->next = 0;
184  shared_->ownBuffer = true;
185  shared_->refcount = 1;
186  }
187 
188  //! Constructor with user-supplied buffer.
189  /*! The user buffer will be used firstly. When it is full, memory pool allocates new chunk with chunk size.
190 
191  The user buffer will not be deallocated when this allocator is destructed.
192 
193  \param buffer User supplied buffer.
194  \param size Size of the buffer in bytes. It must at least larger than sizeof(ChunkHeader).
195  \param chunkSize The size of memory chunk. The default is kDefaultChunkSize.
196  \param baseAllocator The allocator for allocating memory chunks.
197  */
198  MemoryPoolAllocator(void *buffer, size_t size, size_t chunkSize = kDefaultChunkCapacity, BaseAllocator* baseAllocator = 0) :
199  chunk_capacity_(chunkSize),
200  baseAllocator_(baseAllocator),
201  shared_(static_cast<SharedData*>(AlignBuffer(buffer, size)))
202  {
203  RAPIDJSON_ASSERT(size >= SIZEOF_SHARED_DATA + SIZEOF_CHUNK_HEADER);
204  shared_->chunkHead = GetChunkHead(shared_);
205  shared_->chunkHead->capacity = size - SIZEOF_SHARED_DATA - SIZEOF_CHUNK_HEADER;
206  shared_->chunkHead->size = 0;
207  shared_->chunkHead->next = 0;
208  shared_->ownBaseAllocator = 0;
209  shared_->ownBuffer = false;
210  shared_->refcount = 1;
211  }
212 
213  MemoryPoolAllocator(const MemoryPoolAllocator& rhs) RAPIDJSON_NOEXCEPT :
214  chunk_capacity_(rhs.chunk_capacity_),
215  baseAllocator_(rhs.baseAllocator_),
216  shared_(rhs.shared_)
217  {
218  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
219  ++shared_->refcount;
220  }
221  MemoryPoolAllocator& operator=(const MemoryPoolAllocator& rhs) RAPIDJSON_NOEXCEPT
222  {
223  RAPIDJSON_NOEXCEPT_ASSERT(rhs.shared_->refcount > 0);
224  ++rhs.shared_->refcount;
225 
226  this->~MemoryPoolAllocator();
227  baseAllocator_ = rhs.baseAllocator_;
228  chunk_capacity_ = rhs.chunk_capacity_;
229  shared_ = rhs.shared_;
230  return *this;
231  }
232 
233  //! Destructor.
234  /*! This deallocates all memory chunks, excluding the user-supplied buffer.
235  */
236  ~MemoryPoolAllocator() RAPIDJSON_NOEXCEPT {
237  if (shared_->refcount > 1) {
238  --shared_->refcount;
239  return;
240  }
241  Clear();
242  BaseAllocator *a = shared_->ownBaseAllocator;
243  if (shared_->ownBuffer) {
244  baseAllocator_->Free(shared_);
245  }
246  RAPIDJSON_DELETE(a);
247  }
248 
249  //! Deallocates all memory chunks, excluding the first/user one.
250  void Clear() RAPIDJSON_NOEXCEPT {
251  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
252  for (;;) {
253  ChunkHeader* c = shared_->chunkHead;
254  if (!c->next) {
255  break;
256  }
257  shared_->chunkHead = c->next;
258  baseAllocator_->Free(c);
259  }
260  shared_->chunkHead->size = 0;
261  }
262 
263  //! Computes the total capacity of allocated memory chunks.
264  /*! \return total capacity in bytes.
265  */
266  size_t Capacity() const RAPIDJSON_NOEXCEPT {
267  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
268  size_t capacity = 0;
269  for (ChunkHeader* c = shared_->chunkHead; c != 0; c = c->next)
270  capacity += c->capacity;
271  return capacity;
272  }
273 
274  //! Computes the memory blocks allocated.
275  /*! \return total used bytes.
276  */
277  size_t Size() const RAPIDJSON_NOEXCEPT {
278  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
279  size_t size = 0;
280  for (ChunkHeader* c = shared_->chunkHead; c != 0; c = c->next)
281  size += c->size;
282  return size;
283  }
284 
285  //! Whether the allocator is shared.
286  /*! \return true or false.
287  */
288  bool Shared() const RAPIDJSON_NOEXCEPT {
289  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
290  return shared_->refcount > 1;
291  }
292 
293  //! Allocates a memory block. (concept Allocator)
294  void* Malloc(size_t size) {
295  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
296  if (!size)
297  return NULL;
298 
299  size = RAPIDJSON_ALIGN(size);
300  if (RAPIDJSON_UNLIKELY(shared_->chunkHead->size + size > shared_->chunkHead->capacity))
301  if (!AddChunk(chunk_capacity_ > size ? chunk_capacity_ : size))
302  return NULL;
303 
304  void *buffer = GetChunkBuffer(shared_) + shared_->chunkHead->size;
305  shared_->chunkHead->size += size;
306  return buffer;
307  }
308 
309  //! Resizes a memory block (concept Allocator)
310  void* Realloc(void* originalPtr, size_t originalSize, size_t newSize) {
311  if (originalPtr == 0)
312  return Malloc(newSize);
313 
314  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
315  if (newSize == 0)
316  return NULL;
317 
318  originalSize = RAPIDJSON_ALIGN(originalSize);
319  newSize = RAPIDJSON_ALIGN(newSize);
320 
321  // Do not shrink if new size is smaller than original
322  if (originalSize >= newSize)
323  return originalPtr;
324 
325  // Simply expand it if it is the last allocation and there is sufficient space
326  if (originalPtr == GetChunkBuffer(shared_) + shared_->chunkHead->size - originalSize) {
327  size_t increment = static_cast<size_t>(newSize - originalSize);
328  if (shared_->chunkHead->size + increment <= shared_->chunkHead->capacity) {
329  shared_->chunkHead->size += increment;
330  return originalPtr;
331  }
332  }
333 
334  // Realloc process: allocate and copy memory, do not free original buffer.
335  if (void* newBuffer = Malloc(newSize)) {
336  if (originalSize)
337  std::memcpy(newBuffer, originalPtr, originalSize);
338  return newBuffer;
339  }
340  else
341  return NULL;
342  }
343 
344  //! Frees a memory block (concept Allocator)
345  static void Free(void *ptr) RAPIDJSON_NOEXCEPT { (void)ptr; } // Do nothing
346 
347  //! Compare (equality) with another MemoryPoolAllocator
348  bool operator==(const MemoryPoolAllocator& rhs) const RAPIDJSON_NOEXCEPT {
349  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
350  RAPIDJSON_NOEXCEPT_ASSERT(rhs.shared_->refcount > 0);
351  return shared_ == rhs.shared_;
352  }
353  //! Compare (inequality) with another MemoryPoolAllocator
354  bool operator!=(const MemoryPoolAllocator& rhs) const RAPIDJSON_NOEXCEPT {
355  return !operator==(rhs);
356  }
357 
358 private:
359  //! Creates a new chunk.
360  /*! \param capacity Capacity of the chunk in bytes.
361  \return true if success.
362  */
363  bool AddChunk(size_t capacity) {
364  if (!baseAllocator_)
365  shared_->ownBaseAllocator = baseAllocator_ = RAPIDJSON_NEW(BaseAllocator)();
366  if (ChunkHeader* chunk = static_cast<ChunkHeader*>(baseAllocator_->Malloc(SIZEOF_CHUNK_HEADER + capacity))) {
367  chunk->capacity = capacity;
368  chunk->size = 0;
369  chunk->next = shared_->chunkHead;
370  shared_->chunkHead = chunk;
371  return true;
372  }
373  else
374  return false;
375  }
376 
377  static inline void* AlignBuffer(void* buf, size_t &size)
378  {
379  RAPIDJSON_NOEXCEPT_ASSERT(buf != 0);
380  const uintptr_t mask = sizeof(void*) - 1;
381  const uintptr_t ubuf = reinterpret_cast<uintptr_t>(buf);
382  if (RAPIDJSON_UNLIKELY(ubuf & mask)) {
383  const uintptr_t abuf = (ubuf + mask) & ~mask;
384  RAPIDJSON_ASSERT(size >= abuf - ubuf);
385  buf = reinterpret_cast<void*>(abuf);
386  size -= abuf - ubuf;
387  }
388  return buf;
389  }
390 
391  size_t chunk_capacity_; //!< The minimum capacity of chunk when they are allocated.
392  BaseAllocator* baseAllocator_; //!< base allocator for allocating memory chunks.
393  SharedData *shared_; //!< The shared data of the allocator
394 };

397 template<typename T, typename A>
398 inline T* Realloc(A& a, T* old_p, size_t old_n, size_t new_n)
399 {
400  RAPIDJSON_NOEXCEPT_ASSERT(old_n <= SIZE_MAX / sizeof(T) && new_n <= SIZE_MAX / sizeof(T));
401  return static_cast<T*>(a.Realloc(old_p, old_n * sizeof(T), new_n * sizeof(T)));
402 }

404 template<typename T, typename A>
405 inline T *Malloc(A& a, size_t n = 1)
406 {
407  return Realloc<T, A>(a, NULL, 0, n);
408 }

410 template<typename T, typename A>
411 inline void Free(A& a, T *p, size_t n = 1)
412 {
413  static_cast<void>(Realloc<T, A>(a, p, n, 0));
414 }
