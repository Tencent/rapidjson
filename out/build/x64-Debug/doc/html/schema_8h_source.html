191// Specification
192enum SchemaDraft {
193 kDraftUnknown = -1,
194 kDraftNone = 0,
195 kDraft03 = 3,
196 kDraftMin = 4, //!< Current minimum supported draft
197 kDraft04 = 4,
198 kDraft05 = 5,
199 kDraftMax = 5, //!< Current maximum supported draft
200 kDraft06 = 6,
201 kDraft07 = 7,
202 kDraft2019_09 = 8,
203 kDraft2020_12 = 9
204};
205
206enum OpenApiVersion {
207 kVersionUnknown = -1,
208 kVersionNone = 0,
209 kVersionMin = 2, //!< Current minimum supported version
210 kVersion20 = 2,
211 kVersion30 = 3,
212 kVersionMax = 3, //!< Current maximum supported version
213 kVersion31 = 4,
214};
215
216struct Specification {
217 Specification(SchemaDraft d) : draft(d), oapi(kVersionNone) {}
218 Specification(OpenApiVersion o) : oapi(o) {
219 if (oapi == kVersion20) draft = kDraft04;
220 else if (oapi == kVersion30) draft = kDraft05;
221 else if (oapi == kVersion31) draft = kDraft2020_12;
222 else draft = kDraft04;
223 }
224 ~Specification() {}
225 bool IsSupported() const {
226 return ((draft >= kDraftMin && draft <= kDraftMax) && ((oapi == kVersionNone) || (oapi >= kVersionMin && oapi <= kVersionMax)));
227 }
228 SchemaDraft draft;
229 OpenApiVersion oapi;
230};

329///////////////////////////////////////////////////////////////////////////////
330// Hasher
331
332// For comparison of compound value
333template<typename Encoding, typename Allocator>
334class Hasher {
335public:
336 typedef typename Encoding::Ch Ch;
337
338 Hasher(Allocator* allocator = 0, size_t stackCapacity = kDefaultSize) : stack_(allocator, stackCapacity) {}
339
340 bool Null() { return WriteType(kNullType); }
341 bool Bool(bool b) { return WriteType(b ? kTrueType : kFalseType); }
342 bool Int(int i) { Number n; n.u.i = i; n.d = static_cast<double>(i); return WriteNumber(n); }
343 bool Uint(unsigned u) { Number n; n.u.u = u; n.d = static_cast<double>(u); return WriteNumber(n); }
344 bool Int64(int64_t i) { Number n; n.u.i = i; n.d = static_cast<double>(i); return WriteNumber(n); }
345 bool Uint64(uint64_t u) { Number n; n.u.u = u; n.d = static_cast<double>(u); return WriteNumber(n); }
346 bool Double(double d) {
347 Number n;
348 if (d < 0) n.u.i = static_cast<int64_t>(d);
349 else n.u.u = static_cast<uint64_t>(d);
350 n.d = d;
351 return WriteNumber(n);
352 }
353
354 bool RawNumber(const Ch* str, SizeType len, bool) {
355 WriteBuffer(kNumberType, str, len * sizeof(Ch));
356 return true;
357 }
358
359 bool String(const Ch* str, SizeType len, bool) {
360 WriteBuffer(kStringType, str, len * sizeof(Ch));
361 return true;
362 }
363
364 bool StartObject() { return true; }
365 bool Key(const Ch* str, SizeType len, bool copy) { return String(str, len, copy); }
366 bool EndObject(SizeType memberCount) {
367 uint64_t h = Hash(0, kObjectType);
368 uint64_t* kv = stack_.template Pop<uint64_t>(memberCount * 2);
369 for (SizeType i = 0; i < memberCount; i++)
370 h ^= Hash(kv[i * 2], kv[i * 2 + 1]); // Use xor to achieve member order insensitive
371 *stack_.template Push<uint64_t>() = h;
372 return true;
373 }
374
375 bool StartArray() { return true; }
376 bool EndArray(SizeType elementCount) {
377 uint64_t h = Hash(0, kArrayType);
378 uint64_t* e = stack_.template Pop<uint64_t>(elementCount);
379 for (SizeType i = 0; i < elementCount; i++)
380 h = Hash(h, e[i]); // Use hash to achieve element order sensitive
381 *stack_.template Push<uint64_t>() = h;
382 return true;
383 }
384
385 bool IsValid() const { return stack_.GetSize() == sizeof(uint64_t); }
386
387 uint64_t GetHashCode() const {
388 RAPIDJSON_ASSERT(IsValid());
389 return *stack_.template Top<uint64_t>();
390 }
391
392private:
393 static const size_t kDefaultSize = 256;
394 struct Number {
395 union U {
396 uint64_t u;
397 int64_t i;
398 }u;
399 double d;
400 };
401
402 bool WriteType(Type type) { return WriteBuffer(type, 0, 0); }
403
404 bool WriteNumber(const Number& n) { return WriteBuffer(kNumberType, &n, sizeof(n)); }
405
406 bool WriteBuffer(Type type, const void* data, size_t len) {
407 // FNV-1a from http://isthe.com/chongo/tech/comp/fnv/
408 uint64_t h = Hash(RAPIDJSON_UINT64_C2(0x84222325, 0xcbf29ce4), type);
409 const unsigned char* d = static_cast<const unsigned char*>(data);
410 for (size_t i = 0; i < len; i++)
411 h = Hash(h, d[i]);
412 *stack_.template Push<uint64_t>() = h;
413 return true;
414 }
415
416 static uint64_t Hash(uint64_t h, uint64_t d) {
417 static const uint64_t kPrime = RAPIDJSON_UINT64_C2(0x00000100, 0x000001b3);
418 h ^= d;
419 h *= kPrime;
420 return h;
421 }
422
423 Stack<Allocator> stack_;
424};

516///////////////////////////////////////////////////////////////////////////////
517// Schema
518
519template <typename SchemaDocumentType>
520class Schema {
521public:
522 typedef typename SchemaDocumentType::ValueType ValueType;
523 typedef typename SchemaDocumentType::AllocatorType AllocatorType;
524 typedef typename SchemaDocumentType::PointerType PointerType;
525 typedef typename ValueType::EncodingType EncodingType;
526 typedef typename EncodingType::Ch Ch;
527 typedef SchemaValidationContext<SchemaDocumentType> Context;
528 typedef Schema<SchemaDocumentType> SchemaType;
529 typedef GenericValue<EncodingType, AllocatorType> SValue;
530 typedef IValidationErrorHandler<Schema> ErrorHandler;
531 typedef GenericUri<ValueType, AllocatorType> UriType;
532 friend class GenericSchemaDocument<ValueType, AllocatorType>;
533
534 Schema(SchemaDocumentType* schemaDocument, const PointerType& p, const ValueType& value, const ValueType& document, AllocatorType* allocator, const UriType& id = UriType()) :
535 allocator_(allocator),
536 uri_(schemaDocument->GetURI(), *allocator),
537 id_(id, allocator),
538 spec_(schemaDocument->GetSpecification()),
539 pointer_(p, allocator),
540 typeless_(schemaDocument->GetTypeless()),
541 enum_(),
542 enumCount_(),
543 not_(),
544 type_((1 << kTotalSchemaType) - 1), // typeless
545 validatorCount_(),
546 notValidatorIndex_(),
547 properties_(),
548 additionalPropertiesSchema_(),
549 patternProperties_(),
550 patternPropertyCount_(),
551 propertyCount_(),
552 minProperties_(),
553 maxProperties_(SizeType(~0)),
554 additionalProperties_(true),
555 hasDependencies_(),
556 hasRequired_(),
557 hasSchemaDependencies_(),
558 additionalItemsSchema_(),
559 itemsList_(),
560 itemsTuple_(),
561 itemsTupleCount_(),
562 minItems_(),
563 maxItems_(SizeType(~0)),
564 additionalItems_(true),
565 uniqueItems_(false),
566 pattern_(),
567 minLength_(0),
568 maxLength_(~SizeType(0)),
569 exclusiveMinimum_(false),
570 exclusiveMaximum_(false),
571 defaultValueLength_(0),
572 readOnly_(false),
573 writeOnly_(false),
574 nullable_(false)
575 {
576 GenericStringBuffer<EncodingType> sb;
577 p.StringifyUriFragment(sb);
578 RAPIDJSON_SCHEMA_PRINT(Method, "Schema::Schema", sb.GetString(), id.GetString());
579
580 typedef typename ValueType::ConstValueIterator ConstValueIterator;
581 typedef typename ValueType::ConstMemberIterator ConstMemberIterator;
582
583 // PR #1393
584 // Early add this Schema and its $ref(s) in schemaDocument's map to avoid infinite
585 // recursion (with recursive schemas), since schemaDocument->getSchema() is always
586 // checked before creating a new one. Don't cache typeless_, though.
587 if (this != typeless_) {
588 typedef typename SchemaDocumentType::SchemaEntry SchemaEntry;
589 SchemaEntry *entry = schemaDocument->schemaMap_.template Push<SchemaEntry>();
590 new (entry) SchemaEntry(pointer_, this, true, allocator_);
591 schemaDocument->AddSchemaRefs(this);
592 }
593
594 if (!value.IsObject())
595 return;
596
597 // If we have an id property, resolve it with the in-scope id
598 // Not supported for open api 2.0 or 3.0
599 if (spec_.oapi != kVersion20 && spec_.oapi != kVersion30)
600 if (const ValueType* v = GetMember(value, GetIdString())) {
601 if (v->IsString()) {
602 UriType local(*v, allocator);
603 id_ = local.Resolve(id_, allocator);
604 RAPIDJSON_SCHEMA_PRINT(SchemaIds, id.GetString(), v->GetString(), id_.GetString());
605 }
606 }
607
608 if (const ValueType* v = GetMember(value, GetTypeString())) {
609 type_ = 0;
610 if (v->IsString())
611 AddType(*v);
612 else if (v->IsArray())
613 for (ConstValueIterator itr = v->Begin(); itr != v->End(); ++itr)
614 AddType(*itr);
615 }
616
617 if (const ValueType* v = GetMember(value, GetEnumString())) {
618 if (v->IsArray() && v->Size() > 0) {
619 enum_ = static_cast<uint64_t*>(allocator_->Malloc(sizeof(uint64_t) * v->Size()));
620 for (ConstValueIterator itr = v->Begin(); itr != v->End(); ++itr) {
621 typedef Hasher<EncodingType, MemoryPoolAllocator<AllocatorType> > EnumHasherType;
622 char buffer[256u + 24];
623 MemoryPoolAllocator<AllocatorType> hasherAllocator(buffer, sizeof(buffer));
624 EnumHasherType h(&hasherAllocator, 256);
625 itr->Accept(h);
626 enum_[enumCount_++] = h.GetHashCode();
627 }
628 }
629 }
630
631 if (schemaDocument)
632 AssignIfExist(allOf_, *schemaDocument, p, value, GetAllOfString(), document);
633
634 // AnyOf, OneOf, Not not supported for open api 2.0
635 if (schemaDocument && spec_.oapi != kVersion20) {
636 AssignIfExist(anyOf_, *schemaDocument, p, value, GetAnyOfString(), document);
637 AssignIfExist(oneOf_, *schemaDocument, p, value, GetOneOfString(), document);
638
639 if (const ValueType* v = GetMember(value, GetNotString())) {
640 schemaDocument->CreateSchema(&not_, p.Append(GetNotString(), allocator_), *v, document, id_);
641 notValidatorIndex_ = validatorCount_;
642 validatorCount_++;
643 }
644 }
645
646 // Object
647
648 const ValueType* properties = GetMember(value, GetPropertiesString());
649 const ValueType* required = GetMember(value, GetRequiredString());
650 const ValueType* dependencies = GetMember(value, GetDependenciesString());
651 {
652 // Gather properties from properties/required/dependencies
653 SValue allProperties(kArrayType);
654
655 if (properties && properties->IsObject())
656 for (ConstMemberIterator itr = properties->MemberBegin(); itr != properties->MemberEnd(); ++itr)
657 AddUniqueElement(allProperties, itr->name);
658
659 if (required && required->IsArray())
660 for (ConstValueIterator itr = required->Begin(); itr != required->End(); ++itr)
661 if (itr->IsString())
662 AddUniqueElement(allProperties, *itr);
663
664 // Dependencies not supported for open api 2.0 and 3.0
665 if (spec_.oapi != kVersion20 && spec_.oapi != kVersion30)
666 if (dependencies && dependencies->IsObject())
667 for (ConstMemberIterator itr = dependencies->MemberBegin(); itr != dependencies->MemberEnd(); ++itr) {
668 AddUniqueElement(allProperties, itr->name);
669 if (itr->value.IsArray())
670 for (ConstValueIterator i = itr->value.Begin(); i != itr->value.End(); ++i)
671 if (i->IsString())
672 AddUniqueElement(allProperties, *i);
673 }
674
675 if (allProperties.Size() > 0) {
676 propertyCount_ = allProperties.Size();
677 properties_ = static_cast<Property*>(allocator_->Malloc(sizeof(Property) * propertyCount_));
678 for (SizeType i = 0; i < propertyCount_; i++) {
679 new (&properties_[i]) Property();
680 properties_[i].name = allProperties[i];
681 properties_[i].schema = typeless_;
682 }
683 }
684 }
685
686 if (properties && properties->IsObject()) {
687 PointerType q = p.Append(GetPropertiesString(), allocator_);
688 for (ConstMemberIterator itr = properties->MemberBegin(); itr != properties->MemberEnd(); ++itr) {
689 SizeType index;
690 if (FindPropertyIndex(itr->name, &index))
691 schemaDocument->CreateSchema(&properties_[index].schema, q.Append(itr->name, allocator_), itr->value, document, id_);
692 }
693 }
694
695 // PatternProperties not supported for open api 2.0 and 3.0
696 if (spec_.oapi != kVersion20 && spec_.oapi != kVersion30)
697 if (const ValueType* v = GetMember(value, GetPatternPropertiesString())) {
698 PointerType q = p.Append(GetPatternPropertiesString(), allocator_);
699 patternProperties_ = static_cast<PatternProperty*>(allocator_->Malloc(sizeof(PatternProperty) * v->MemberCount()));
700 patternPropertyCount_ = 0;
701
702 for (ConstMemberIterator itr = v->MemberBegin(); itr != v->MemberEnd(); ++itr) {
703 new (&patternProperties_[patternPropertyCount_]) PatternProperty();
704 PointerType r = q.Append(itr->name, allocator_);
705 patternProperties_[patternPropertyCount_].pattern = CreatePattern(itr->name, schemaDocument, r);
706 schemaDocument->CreateSchema(&patternProperties_[patternPropertyCount_].schema, r, itr->value, document, id_);
707 patternPropertyCount_++;
708 }
709 }
710
711 if (required && required->IsArray())
712 for (ConstValueIterator itr = required->Begin(); itr != required->End(); ++itr)
713 if (itr->IsString()) {
714 SizeType index;
715 if (FindPropertyIndex(*itr, &index)) {
716 properties_[index].required = true;
717 hasRequired_ = true;
718 }
719 }
720
721 // Dependencies not supported for open api 2.0 and 3.0
722 if (spec_.oapi != kVersion20 && spec_.oapi != kVersion30)
723 if (dependencies && dependencies->IsObject()) {
724 PointerType q = p.Append(GetDependenciesString(), allocator_);
725 hasDependencies_ = true;
726 for (ConstMemberIterator itr = dependencies->MemberBegin(); itr != dependencies->MemberEnd(); ++itr) {
727 SizeType sourceIndex;
728 if (FindPropertyIndex(itr->name, &sourceIndex)) {
729 if (itr->value.IsArray()) {
730 properties_[sourceIndex].dependencies = static_cast<bool*>(allocator_->Malloc(sizeof(bool) * propertyCount_));
731 std::memset(properties_[sourceIndex].dependencies, 0, sizeof(bool)* propertyCount_);
732 for (ConstValueIterator targetItr = itr->value.Begin(); targetItr != itr->value.End(); ++targetItr) {
733 SizeType targetIndex;
734 if (FindPropertyIndex(*targetItr, &targetIndex))
735 properties_[sourceIndex].dependencies[targetIndex] = true;
736 }
737 }
738 else if (itr->value.IsObject()) {
739 hasSchemaDependencies_ = true;
740 schemaDocument->CreateSchema(&properties_[sourceIndex].dependenciesSchema, q.Append(itr->name, allocator_), itr->value, document, id_);
741 properties_[sourceIndex].dependenciesValidatorIndex = validatorCount_;
742 validatorCount_++;
743 }
744 }
745 }
746 }
747
748 if (const ValueType* v = GetMember(value, GetAdditionalPropertiesString())) {
749 if (v->IsBool())
750 additionalProperties_ = v->GetBool();
751 else if (v->IsObject())
752 schemaDocument->CreateSchema(&additionalPropertiesSchema_, p.Append(GetAdditionalPropertiesString(), allocator_), *v, document, id_);
753 }
754
755 AssignIfExist(minProperties_, value, GetMinPropertiesString());
756 AssignIfExist(maxProperties_, value, GetMaxPropertiesString());
757
758 // Array
759 if (const ValueType* v = GetMember(value, GetItemsString())) {
760 PointerType q = p.Append(GetItemsString(), allocator_);
761 if (v->IsObject()) // List validation
762 schemaDocument->CreateSchema(&itemsList_, q, *v, document, id_);
763 else if (v->IsArray()) { // Tuple validation
764 itemsTuple_ = static_cast<const Schema**>(allocator_->Malloc(sizeof(const Schema*) * v->Size()));
765 SizeType index = 0;
766 for (ConstValueIterator itr = v->Begin(); itr != v->End(); ++itr, index++)
767 schemaDocument->CreateSchema(&itemsTuple_[itemsTupleCount_++], q.Append(index, allocator_), *itr, document, id_);
768 }
769 }
770
771 AssignIfExist(minItems_, value, GetMinItemsString());
772 AssignIfExist(maxItems_, value, GetMaxItemsString());
773
774 // AdditionalItems not supported for openapi 2.0 and 3.0
775 if (spec_.oapi != kVersion20 && spec_.oapi != kVersion30)
776 if (const ValueType* v = GetMember(value, GetAdditionalItemsString())) {
777 if (v->IsBool())
778 additionalItems_ = v->GetBool();
779 else if (v->IsObject())
780 schemaDocument->CreateSchema(&additionalItemsSchema_, p.Append(GetAdditionalItemsString(), allocator_), *v, document, id_);
781 }
782
783 AssignIfExist(uniqueItems_, value, GetUniqueItemsString());
784
785 // String
786 AssignIfExist(minLength_, value, GetMinLengthString());
787 AssignIfExist(maxLength_, value, GetMaxLengthString());
788
789 if (const ValueType* v = GetMember(value, GetPatternString()))
790 pattern_ = CreatePattern(*v, schemaDocument, p.Append(GetPatternString(), allocator_));
791
792 // Number
793 if (const ValueType* v = GetMember(value, GetMinimumString()))
794 if (v->IsNumber())
795 minimum_.CopyFrom(*v, *allocator_);
796
797 if (const ValueType* v = GetMember(value, GetMaximumString()))
798 if (v->IsNumber())
799 maximum_.CopyFrom(*v, *allocator_);
800
801 AssignIfExist(exclusiveMinimum_, value, GetExclusiveMinimumString());
802 AssignIfExist(exclusiveMaximum_, value, GetExclusiveMaximumString());
803
804 if (const ValueType* v = GetMember(value, GetMultipleOfString()))
805 if (v->IsNumber() && v->GetDouble() > 0.0)
806 multipleOf_.CopyFrom(*v, *allocator_);
807
808 // Default
809 if (const ValueType* v = GetMember(value, GetDefaultValueString()))
810 if (v->IsString())
811 defaultValueLength_ = v->GetStringLength();
812
813 // ReadOnly - open api only (until draft 7 supported)
814 // WriteOnly - open api 3 only (until draft 7 supported)
815 // Both can't be true
816 if (spec_.oapi != kVersionNone)
817 AssignIfExist(readOnly_, value, GetReadOnlyString());
818 if (spec_.oapi >= kVersion30)
819 AssignIfExist(writeOnly_, value, GetWriteOnlyString());
820 if (readOnly_ && writeOnly_)
821 schemaDocument->SchemaError(kSchemaErrorReadOnlyAndWriteOnly, p);
822
823 // Nullable - open api 3 only
824 // If true add 'null' as allowable type
825 if (spec_.oapi >= kVersion30) {
826 AssignIfExist(nullable_, value, GetNullableString());
827 if (nullable_)
828 AddType(GetNullString());
829 }

901 RAPIDJSON_FORCEINLINE bool EndValue(Context& context) const {
902 RAPIDJSON_SCHEMA_PRINT(Method, "Schema::EndValue");
903 // Only check pattern properties if we have validators
904 if (context.patternPropertiesValidatorCount > 0) {
905 bool otherValid = false;
906 SizeType count = context.patternPropertiesValidatorCount;
907 if (context.objectPatternValidatorType != Context::kPatternValidatorOnly)
908 otherValid = context.patternPropertiesValidators[--count]->IsValid();
909
910 bool patternValid = true;
911 for (SizeType i = 0; i < count; i++)
912 if (!context.patternPropertiesValidators[i]->IsValid()) {
913 patternValid = false;
914 break;
915 }
916
917 if (context.objectPatternValidatorType == Context::kPatternValidatorOnly) {
918 if (!patternValid) {
919 context.error_handler.PropertyViolations(context.patternPropertiesValidators, count);
920 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorPatternProperties);
921 }
922 }
923 else if (context.objectPatternValidatorType == Context::kPatternValidatorWithProperty) {
924 if (!patternValid || !otherValid) {
925 context.error_handler.PropertyViolations(context.patternPropertiesValidators, count + 1);
926 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorPatternProperties);
927 }
928 }
929 else if (!patternValid && !otherValid) { // kPatternValidatorWithAdditionalProperty)
930 context.error_handler.PropertyViolations(context.patternPropertiesValidators, count + 1);
931 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorPatternProperties);
932 }
933 }
934
935 // For enums only check if we have a hasher
936 if (enum_ && context.hasher) {
937 const uint64_t h = context.factory.GetHashCode(context.hasher);
938 for (SizeType i = 0; i < enumCount_; i++)
939 if (enum_[i] == h)
940 goto foundEnum;
941 context.error_handler.DisallowedValue(kValidateErrorEnum);
942 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorEnum);
943 foundEnum:;
944 }
945
946 // Only check allOf etc if we have validators
947 if (context.validatorCount > 0) {
948 if (allOf_.schemas)
949 for (SizeType i = allOf_.begin; i < allOf_.begin + allOf_.count; i++)
950 if (!context.validators[i]->IsValid()) {
951 context.error_handler.NotAllOf(&context.validators[allOf_.begin], allOf_.count);
952 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorAllOf);
953 }
954
955 if (anyOf_.schemas) {
956 for (SizeType i = anyOf_.begin; i < anyOf_.begin + anyOf_.count; i++)
957 if (context.validators[i]->IsValid())
958 goto foundAny;
959 context.error_handler.NoneOf(&context.validators[anyOf_.begin], anyOf_.count);
960 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorAnyOf);
961 foundAny:;
962 }
963
964 if (oneOf_.schemas) {
965 bool oneValid = false;
966 SizeType firstMatch = 0;
967 for (SizeType i = oneOf_.begin; i < oneOf_.begin + oneOf_.count; i++)
968 if (context.validators[i]->IsValid()) {
969 if (oneValid) {
970 context.error_handler.MultipleOneOf(firstMatch, i - oneOf_.begin);
971 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorOneOfMatch);
972 } else {
973 oneValid = true;
974 firstMatch = i - oneOf_.begin;
975 }
976 }
977 if (!oneValid) {
978 context.error_handler.NotOneOf(&context.validators[oneOf_.begin], oneOf_.count);
979 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorOneOf);
980 }
981 }
982
983 if (not_ && context.validators[notValidatorIndex_]->IsValid()) {
984 context.error_handler.Disallowed();
985 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorNot);
986 }
987 }
988
989 return true;
990 }

1108
1109 if (patternProperties_) {
1110 context.patternPropertiesSchemaCount = 0;

1113 context.patternPropertiesSchemas[context.patternPropertiesSchemaCount++] = patternProperties_[i].schema;
1114 context.valueSchema = typeless_;
1115 }
1116 }
1117
1118 SizeType index = 0;
1119 if (FindPropertyIndex(ValueType(str, len).Move(), &index)) {
1120 if (context.patternPropertiesSchemaCount > 0) {
1121 context.patternPropertiesSchemas[context.patternPropertiesSchemaCount++] = properties_[index].schema;
1122 context.valueSchema = typeless_;

1124 }
1125 else
1126 context.valueSchema = properties_[index].schema;

1159 bool EndObject(Context& context, SizeType memberCount) const {
1160 RAPIDJSON_SCHEMA_PRINT(Method, "Schema::EndObject");
1161 if (hasRequired_) {
1162 context.error_handler.StartMissingProperties();
1163 for (SizeType index = 0; index < propertyCount_; index++)
1164 if (properties_[index].required && !context.propertyExist[index])
1165 if (properties_[index].schema->defaultValueLength_ == 0 )
1166 context.error_handler.AddMissingProperty(properties_[index].name);
1167 if (context.error_handler.EndMissingProperties())
1168 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorRequired);
1169 }
1170
1171 if (memberCount < minProperties_) {
1172 context.error_handler.TooFewProperties(memberCount, minProperties_);
1173 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorMinProperties);
1174 }
1175
1176 if (memberCount > maxProperties_) {
1177 context.error_handler.TooManyProperties(memberCount, maxProperties_);
1178 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorMaxProperties);
1179 }
1180
1181 if (hasDependencies_) {
1182 context.error_handler.StartDependencyErrors();
1183 for (SizeType sourceIndex = 0; sourceIndex < propertyCount_; sourceIndex++) {
1184 const Property& source = properties_[sourceIndex];
1185 if (context.propertyExist[sourceIndex]) {
1186 if (source.dependencies) {
1187 context.error_handler.StartMissingDependentProperties();
1188 for (SizeType targetIndex = 0; targetIndex < propertyCount_; targetIndex++)
1189 if (source.dependencies[targetIndex] && !context.propertyExist[targetIndex])
1190 context.error_handler.AddMissingDependentProperty(properties_[targetIndex].name);
1191 context.error_handler.EndMissingDependentProperties(source.name);
1192 }
1193 else if (source.dependenciesSchema) {
1194 ISchemaValidator* dependenciesValidator = context.validators[source.dependenciesValidatorIndex];
1195 if (!dependenciesValidator->IsValid())
1196 context.error_handler.AddDependencySchemaError(source.name, dependenciesValidator);
1197 }
1198 }
1199 }
1200 if (context.error_handler.EndDependencyErrors())
1201 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorDependencies);
1202 }
1203
1204 return true;
1205 }

1459 // Creates parallel validators for allOf, anyOf, oneOf, not and schema dependencies, if required.
1460 // Also creates a hasher for enums and array uniqueness, if required.
1461 // Also a useful place to add type-independent error checks.
1462 bool CreateParallelValidator(Context& context) const {
1463 if (enum_ || context.arrayUniqueness)
1464 context.hasher = context.factory.CreateHasher();
1465
1466 if (validatorCount_) {
1467 RAPIDJSON_ASSERT(context.validators == 0);
1468 context.validators = static_cast<ISchemaValidator**>(context.factory.MallocState(sizeof(ISchemaValidator*) * validatorCount_));
1469 std::memset(context.validators, 0, sizeof(ISchemaValidator*) * validatorCount_);
1470 context.validatorCount = validatorCount_;
1471
1472 // Always return after first failure for these sub-validators
1473 if (allOf_.schemas)
1474 CreateSchemaValidators(context, allOf_, false);
1475
1476 if (anyOf_.schemas)
1477 CreateSchemaValidators(context, anyOf_, false);
1478
1479 if (oneOf_.schemas)
1480 CreateSchemaValidators(context, oneOf_, false);
1481
1482 if (not_)
1483 context.validators[notValidatorIndex_] = context.factory.CreateSchemaValidator(*not_, false);
1484
1485 if (hasSchemaDependencies_) {
1486 for (SizeType i = 0; i < propertyCount_; i++)
1487 if (properties_[i].dependenciesSchema)
1488 context.validators[properties_[i].dependenciesValidatorIndex] = context.factory.CreateSchemaValidator(*properties_[i].dependenciesSchema, false);
1489 }
1490 }
1491
1492 // Add any other type-independent checks here
1493 if (readOnly_ && (context.flags & kValidateWriteFlag)) {
1494 context.error_handler.DisallowedWhenWriting();
1495 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorReadOnly);
1496 }
1497 if (writeOnly_ && (context.flags & kValidateReadFlag)) {
1498 context.error_handler.DisallowedWhenReading();
1499 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorWriteOnly);
1500 }
1501
1502 return true;
1503 }

1532 bool CheckInt(Context& context, int64_t i) const {
1533 if (!(type_ & ((1 << kIntegerSchemaType) | (1 << kNumberSchemaType)))) {
1534 DisallowedType(context, GetIntegerString());
1535 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorType);

1539 if (minimum_.IsInt64()) {
1540 if (exclusiveMinimum_ ? i <= minimum_.GetInt64() : i < minimum_.GetInt64()) {
1541 context.error_handler.BelowMinimum(i, minimum_, exclusiveMinimum_);
1542 RAPIDJSON_INVALID_KEYWORD_RETURN(exclusiveMinimum_ ? kValidateErrorExclusiveMinimum : kValidateErrorMinimum);
1543 }

1546 context.error_handler.BelowMinimum(i, minimum_, exclusiveMinimum_);
1547 RAPIDJSON_INVALID_KEYWORD_RETURN(exclusiveMinimum_ ? kValidateErrorExclusiveMinimum : kValidateErrorMinimum); // i <= max(int64_t) < minimum.GetUint64()
1548 }
1549 else if (!CheckDoubleMinimum(context, static_cast<double>(i)))
1550 return false;
1551 }
1552
1553 if (!maximum_.IsNull()) {
1554 if (maximum_.IsInt64()) {
1555 if (exclusiveMaximum_ ? i >= maximum_.GetInt64() : i > maximum_.GetInt64()) {
1556 context.error_handler.AboveMaximum(i, maximum_, exclusiveMaximum_);
1557 RAPIDJSON_INVALID_KEYWORD_RETURN(exclusiveMaximum_ ? kValidateErrorExclusiveMaximum : kValidateErrorMaximum);
1558 }
1559 }
1560 else if (maximum_.IsUint64()) { }
1561 /* do nothing */ // i <= max(int64_t) < maximum_.GetUint64()
1562 else if (!CheckDoubleMaximum(context, static_cast<double>(i)))
1563 return false;
1564 }
1565
1566 if (!multipleOf_.IsNull()) {
1567 if (multipleOf_.IsUint64()) {
1568 if (static_cast<uint64_t>(i >= 0 ? i : -i) % multipleOf_.GetUint64() != 0) {
1569 context.error_handler.NotMultipleOf(i, multipleOf_);
1570 RAPIDJSON_INVALID_KEYWORD_RETURN(kValidateErrorMultipleOf);
1571 }
1572 }
1573 else if (!CheckDoubleMultipleOf(context, static_cast<double>(i)))
1574 return false;
1575 }
1576
1577 return true;
1578 }
