27 //! Writer with indentation and spacing.
28 /*!
29  \tparam OutputStream Type of ouptut os.
30  \tparam SourceEncoding Encoding of source string.
31  \tparam TargetEncoding Encoding of output stream.
32  \tparam StackAllocator Type of allocator for allocating memory of stack.
33 */
34 template<typename OutputStream, typename SourceEncoding = UTF8<>, typename TargetEncoding = UTF8<>, typename StackAllocator = CrtAllocator, unsigned writeFlags = kWriteDefaultFlags>
35 class PrettyWriter : public Writer<OutputStream, SourceEncoding, TargetEncoding, StackAllocator, writeFlags> {

52  //! Set custom indentation.
53  /*! \param indentChar Character for indentation. Must be whitespace character (' ', '\\t', '\\n', '\\r').
54  \param indentCharCount Number of indent characters for each indentation level.
55  \note The default indentation is 4 spaces.
56  */
57  PrettyWriter& SetIndent(Ch indentChar, unsigned indentCharCount) {
58  RAPIDJSON_ASSERT(indentChar == ' ' || indentChar == '\t' || indentChar == '\n' || indentChar == '\r');

61  return *this;

89  bool StartObject() {

97  bool EndObject(SizeType memberCount = 0) {
98  (void)memberCount;
99  RAPIDJSON_ASSERT(Base::level_stack_.GetSize() >= sizeof(typename Base::Level));
100  RAPIDJSON_ASSERT(!Base::level_stack_.template Top<typename Base::Level>()->inArray);
101  bool empty = Base::level_stack_.template Pop<typename Base::Level>(1)->valueCount == 0;
102 
103  if (!empty) {
104  Base::os_->Put('\n');
105  WriteIndent();
106  }
107  bool ret = Base::WriteEndObject();
108  (void)ret;
109  RAPIDJSON_ASSERT(ret == true);
110  if (Base::level_stack_.Empty()) // end of json text
111  Base::os_->Flush();
112  return true;
113  }

149 protected:
150  void PrettyPrefix(Type type) {
151  (void)type;
152  if (Base::level_stack_.GetSize() != 0) { // this value is not at root
153  typename Base::Level* level = Base::level_stack_.template Top<typename Base::Level>();
154 
155  if (level->inArray) {
156  if (level->valueCount > 0) {
157  Base::os_->Put(','); // add comma if it is not the first element in array
158  Base::os_->Put('\n');
159  }
160  else
161  Base::os_->Put('\n');
162  WriteIndent();
163  }
164  else { // in object
165  if (level->valueCount > 0) {
166  if (level->valueCount % 2 == 0) {
167  Base::os_->Put(',');
168  Base::os_->Put('\n');
169  }
170  else {
171  Base::os_->Put(':');
172  Base::os_->Put(' ');
173  }
174  }
175  else
176  Base::os_->Put('\n');
177 
178  if (level->valueCount % 2 == 0)
179  WriteIndent();
180  }
181  if (!level->inArray && level->valueCount % 2 == 0)
182  RAPIDJSON_ASSERT(type == kStringType); // if it's in object, then even number should be a name
183  level->valueCount++;
184  }
185  else {
186  RAPIDJSON_ASSERT(!Base::hasRoot_); // Should only has one and only one root.
187  Base::hasRoot_ = true;
188  }
189  }
190 
191  void WriteIndent() {
192  size_t count = (Base::level_stack_.GetSize() / sizeof(typename Base::Level)) * indentCharCount_;
193  PutN(*Base::os_, static_cast<typename TargetEncoding::Ch>(indentChar_), count);
194  }

203 };
