30 //! Wrapper of \c std::basic_istream into RapidJSON's Stream concept.
31 /*!
32  The classes can be wrapped including but not limited to:
33 
34  - \c std::istringstream
35  - \c std::stringstream
36  - \c std::wistringstream
37  - \c std::wstringstream
38  - \c std::ifstream
39  - \c std::fstream
40  - \c std::wifstream
41  - \c std::wfstream
42 
43  \tparam StreamType Class derived from \c std::basic_istream.
44 */
45 
46 template <typename StreamType>
47 class BasicIStreamWrapper {
48 public:
49  typedef typename StreamType::char_type Ch;
50  BasicIStreamWrapper(StreamType& stream) : stream_(stream), count_(), peekBuffer_() {}
51 
52  Ch Peek() const {
53  typename StreamType::int_type c = stream_.peek();
54  return RAPIDJSON_LIKELY(c != StreamType::traits_type::eof()) ? static_cast<Ch>(c) : '\0';
55  }
56 
57  Ch Take() {
58  typename StreamType::int_type c = stream_.get();
59  if (RAPIDJSON_LIKELY(c != StreamType::traits_type::eof())) {
60  count_++;
61  return static_cast<Ch>(c);
62  }
63  else
64  return '\0';
65  }
66 
67  // tellg() may return -1 when failed. So we count by ourself.
68  size_t Tell() const { return count_; }
69 
70  Ch* PutBegin() { RAPIDJSON_ASSERT(false); return 0; }
71  void Put(Ch) { RAPIDJSON_ASSERT(false); }
72  void Flush() { RAPIDJSON_ASSERT(false); }
73  size_t PutEnd(Ch*) { RAPIDJSON_ASSERT(false); return 0; }
74 
75  // For encoding detection only.
76  const Ch* Peek4() const {
77  RAPIDJSON_ASSERT(sizeof(Ch) == 1); // Only usable for byte stream.
78  int i;
79  bool hasError = false;
80  for (i = 0; i < 4; ++i) {
81  typename StreamType::int_type c = stream_.get();
82  if (c == StreamType::traits_type::eof()) {
83  hasError = true;
84  stream_.clear();
85  break;
86  }
87  peekBuffer_[i] = static_cast<Ch>(c);
88  }
89  for (--i; i >= 0; --i)
90  stream_.putback(peekBuffer_[i]);
91  return !hasError ? peekBuffer_ : 0;
92  }
93 
94 private:
95  BasicIStreamWrapper(const BasicIStreamWrapper&);
96  BasicIStreamWrapper& operator=(const BasicIStreamWrapper&);
97 
98  StreamType& stream_;
99  size_t count_; //!< Number of characters read. Note:
100  mutable Ch peekBuffer_[4];
101 };
