225 //! Reference to a constant string (not taking a copy)
226 /*!
227  \tparam CharType character type of the string
228 
229  This helper class is used to automatically infer constant string
230  references for string literals, especially from \c const \b (!)
231  character arrays.
232 
233  The main use is for creating JSON string values without copying the
234  source string via an \ref Allocator. This requires that the referenced
235  string pointers have a sufficient lifetime, which exceeds the lifetime
236  of the associated GenericValue.
237 
238  \b Example
239  \code
240  Value v("foo"); // ok, no need to copy & calculate length
241  const char foo[] = "foo";
242  v.SetString(foo); // ok
243 
244  const char* bar = foo;
245  // Value x(bar); // not ok, can't rely on bar's lifetime
246  Value x(StringRef(bar)); // lifetime explicitly guaranteed by user
247  Value y(StringRef(bar, 3)); // ok, explicitly pass length
248  \endcode
249 
250  \see StringRef, GenericValue::SetString
251 */
252 template<typename CharType>
253 struct GenericStringRef {
254  typedef CharType Ch; //!< character type of the string
255 
256  //! Create string reference from \c const character array
257  /*!
258  This constructor implicitly creates a constant string reference from
259  a \c const character array. It has better performance than
260  \ref StringRef(const CharType*) by inferring the string \ref length
261  from the array length, and also supports strings containing null
262  characters.
263 
264  \tparam N length of the string, automatically inferred
265 
266  \param str Constant character array, lifetime assumed to be longer
267  than the use of the string in e.g. a GenericValue
268 
269  \post \ref s == str
270 
271  \note Constant complexity.
272  \note There is a hidden, private overload to disallow references to
273  non-const character arrays to be created via this constructor.
274  By this, e.g. function-scope arrays used to be filled via
275  \c snprintf are excluded from consideration.
276  In such cases, the referenced string should be \b copied to the
277  GenericValue instead.
278  */
279  template<SizeType N>
280  GenericStringRef(const CharType (&str)[N])
281  : s(str), length(N-1) {}
282 
283  //! Explicitly create string reference from \c const character pointer
284  /*!
285  This constructor can be used to \b explicitly create a reference to
286  a constant string pointer.
287 
288  \see StringRef(const CharType*)
289 
290  \param str Constant character pointer, lifetime assumed to be longer
291  than the use of the string in e.g. a GenericValue
292 
293  \post \ref s == str
294 
295  \note There is a hidden, private overload to disallow references to
296  non-const character arrays to be created via this constructor.
297  By this, e.g. function-scope arrays used to be filled via
298  \c snprintf are excluded from consideration.
299  In such cases, the referenced string should be \b copied to the
300  GenericValue instead.
301  */
302  explicit GenericStringRef(const CharType* str)
303  : s(str), length(internal::StrLen(str)){}
304 
305  //! Create constant string reference from pointer and length
306  /*! \param str constant string, lifetime assumed to be longer than the use of the string in e.g. a GenericValue
307  \param len length of the string, excluding the trailing NULL terminator
308 
309  \post \ref s == str && \ref length == len
310  \note Constant complexity.
311  */
312  GenericStringRef(const CharType* str, SizeType len)
313  : s(str), length(len) { RAPIDJSON_ASSERT(s != NULL); }
314 
315  //! implicit conversion to plain CharType pointer
316  operator const Ch *() const { return s; }
317 
318  const Ch* const s; //!< plain CharType pointer
319  const SizeType length; //!< length of the string (excluding the trailing NULL terminator)
320 
321 private:
322  //! Disallow copy-assignment
323  GenericStringRef operator=(const GenericStringRef&);
324  //! Disallow construction from non-const array
325  template<SizeType N>
326  GenericStringRef(CharType (&str)[N]) /* = delete */;
327 };

638  return false;
639 
640  switch (GetType()) {
641  case kObjectType: // Warning: O(n^2) inner-loop
642  if (data_.o.size != rhs.data_.o.size)

644  for (ConstMemberIterator lhsMemberItr = MemberBegin(); lhsMemberItr != MemberEnd(); ++lhsMemberItr) {
645  ConstMemberIterator rhsMemberItr = rhs.FindMember(lhsMemberItr->name);
646  if (rhsMemberItr == rhs.MemberEnd() || lhsMemberItr->value != rhsMemberItr->value)
647  return false;

649  return true;
650 
651  case kArrayType:
652  if (data_.a.size != rhs.data_.a.size)
653  return false;
654  for (SizeType i = 0; i < data_.a.size; i++)
655  if ((*this)[i] != rhs[i])
656  return false;
657  return true;
658 
659  case kStringType:
660  return StringEqual(rhs);
661 
662  case kNumberType:
663  if (IsDouble() || rhs.GetDouble())
664  return GetDouble() == rhs.GetDouble(); // May convert one operand from integer to double.
665  else
666  return data_.n.u64 == rhs.data_.n.u64;
667 
668  default: // kTrueType, kFalseType, kNullType

670  }
671  }
