216 //! Reference to a constant string (not taking a copy)
217 /*!
218  \tparam CharType character type of the string
219 
220  This helper class is used to automatically infer constant string
221  references for string literals, especially from \c const \b (!)
222  character arrays.
223 
224  The main use is for creating JSON string values without copying the
225  source string via an \ref Allocator. This requires that the referenced
226  string pointers have a sufficient lifetime, which exceeds the lifetime
227  of the associated GenericValue.
228 
229  \b Example
230  \code
231  Value v("foo"); // ok, no need to copy & calculate length
232  const char foo[] = "foo";
233  v.SetString(foo); // ok
234 
235  const char* bar = foo;
236  // Value x(bar); // not ok, can't rely on bar's lifetime
237  Value x(StringRef(bar)); // lifetime explicitly guaranteed by user
238  Value y(StringRef(bar, 3)); // ok, explicitly pass length
239  \endcode
240 
241  \see StringRef, GenericValue::SetString
242 */
243 template<typename CharType>
244 struct GenericStringRef {
245  typedef CharType Ch; //!< character type of the string
246 
247  //! Create string reference from \c const character array
248 #ifndef __clang__ // -Wdocumentation
249  /*!
250  This constructor implicitly creates a constant string reference from
251  a \c const character array. It has better performance than
252  \ref StringRef(const CharType*) by inferring the string \ref length
253  from the array length, and also supports strings containing null
254  characters.
255 
256  \tparam N length of the string, automatically inferred
257 
258  \param str Constant character array, lifetime assumed to be longer
259  than the use of the string in e.g. a GenericValue
260 
261  \post \ref s == str
262 
263  \note Constant complexity.
264  \note There is a hidden, private overload to disallow references to
265  non-const character arrays to be created via this constructor.
266  By this, e.g. function-scope arrays used to be filled via
267  \c snprintf are excluded from consideration.
268  In such cases, the referenced string should be \b copied to the
269  GenericValue instead.
270  */
271 #endif
272  template<SizeType N>
273  GenericStringRef(const CharType (&str)[N]) RAPIDJSON_NOEXCEPT
274  : s(str), length(N-1) {}
275 
276  //! Explicitly create string reference from \c const character pointer
277 #ifndef __clang__ // -Wdocumentation
278  /*!
279  This constructor can be used to \b explicitly create a reference to
280  a constant string pointer.
281 
282  \see StringRef(const CharType*)
283 
284  \param str Constant character pointer, lifetime assumed to be longer
285  than the use of the string in e.g. a GenericValue
286 
287  \post \ref s == str
288 
289  \note There is a hidden, private overload to disallow references to
290  non-const character arrays to be created via this constructor.
291  By this, e.g. function-scope arrays used to be filled via
292  \c snprintf are excluded from consideration.
293  In such cases, the referenced string should be \b copied to the
294  GenericValue instead.
295  */
296 #endif
297  explicit GenericStringRef(const CharType* str)
298  : s(str), length(internal::StrLen(str)){ RAPIDJSON_ASSERT(s != NULL); }
299 
300  //! Create constant string reference from pointer and length
301 #ifndef __clang__ // -Wdocumentation
302  /*! \param str constant string, lifetime assumed to be longer than the use of the string in e.g. a GenericValue
303  \param len length of the string, excluding the trailing NULL terminator
304 
305  \post \ref s == str && \ref length == len
306  \note Constant complexity.
307  */
308 #endif
309  GenericStringRef(const CharType* str, SizeType len)
310  : s(str), length(len) { RAPIDJSON_ASSERT(s != NULL); }
311 
312  //! implicit conversion to plain CharType pointer
313  operator const Ch *() const { return s; }
314 
315  const Ch* const s; //!< plain CharType pointer
316  const SizeType length; //!< length of the string (excluding the trailing NULL terminator)
317 
318 private:
319  //! Disallow construction from non-const array
320  template<SizeType N>
321  GenericStringRef(CharType (&str)[N]) /* = delete */;
322 };

686  //! Equal-to operator
687  /*!
688  \note If an object contains duplicated named member, comparing equality with any object is always \c false.
689  \note Linear time complexity (number of all values in the subtree and total lengths of all strings).
690  */
691  template <typename SourceAllocator>
692  bool operator==(const GenericValue<Encoding, SourceAllocator>& rhs) const {
693  typedef GenericValue<Encoding, SourceAllocator> RhsType;
694  if (GetType() != rhs.GetType())

703  if (rhsMemberItr == rhs.MemberEnd() || lhsMemberItr->value != rhsMemberItr->value)

708  case kArrayType:
709  if (data_.a.size != rhs.data_.a.size)
710  return false;

719  case kNumberType:

723  return a >= b && a <= b; // Prevent -Wfloat-equal
724  }
725  else
726  return data_.n.u64 == rhs.data_.n.u64;
727 
728  default:
729  return true;
730  }
